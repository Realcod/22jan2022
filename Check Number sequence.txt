/*
You are given S a sequence of n integers i.e. S = s1, s2, ..., sn. Compute if it is possible to split S into two parts :
s1, s2, ..., si and si+1, si+2, â€¦.., sn (0 <= i <= n) in such a way that the first part is strictly decreasing while
the second is strictly increasing one.
*/
#include<iostream>
#include<math.h>
using namespace std;

int main(){
	int n,i=2,prev,curr;
	bool isdec=true;
	cout<<"Enter n"<<endl;
	cin>>n;
	
	cout<<"Enter previous number"<<endl;
	cin>>prev;
	
	while(i<=n){
		cout<<"Enter current number"<<endl;
		cin>>curr;
		
		if(prev==curr){
			cout<<"false";
			return 0;
		}
		else if(curr>prev){
			if(isdec){
				isdec=false;
			}
			prev=curr;
		}
		else if(curr<prev){
			if(!isdec){
				cout<<"false";
				return 0;
			}
			prev=curr;
		}
		i++;
	}
        cout<<"true";
}
